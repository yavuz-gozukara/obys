function initQRScanner() { const scanner = new Html5QrcodeScanner('qr-reader', { fps: 10, qrbox: 250 }); scanner.render(async (decodedText) => { try { const response = await fetch('/verify_qr', { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify({ qr_data: decodedText }) }); const result = await response.json(); if (result.valid) { alert('Yoklama başarıyla kaydedildi!'); window.location.href = "{{ url_for('student_dashboard') }}"; } else { alert(`Hata: ${result.message}`); } } catch (error) { alert('QR kod işlenirken hata oluştu: ' + error); } }); }
